CREATE VIEW DEPTINFO (NAME,COUNT_EMP,SUM_SAL)AS
SELECT D.DNAME, COUNT(*),SUM(SALARY)
FROM DEPARTMENT D INNER JOIN EMPLOYEE E
ON E.DNO = D.DNUM
GROUP BY D.DNAME;

SELECT * FROM DEPTINFO;
